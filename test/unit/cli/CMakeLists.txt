    add_executable(${TEST} ${SRC} engine_empty.c)

    target_link_libraries(

        states_sequencer_testing.o
        flags_prng_testing.o
        flags_sequencer_testing.o
        memmgr_runtime.o
        memmgr_user.o)

endforeach()
